## Image Loading Pattern (Phase 5 reference)

Standard preload pattern for vanilla JS. All sprites must load before the game loop starts.

    const sprites = {};

    function loadSprites(onComplete) {
      const toLoad = {
        catIdle:       'PixelArt/Cats/cat_idle.png',
        catThrow:      'PixelArt/Cats/cat_throw.png',
        cloudsBright:  'PixelArt/Paralax/Layer1/Clouds_bright.png',
        cloudsDark:    'PixelArt/Paralax/Layer1/Clouds_dark.png',
        bgDay:         'PixelArt/Paralax/Layer1/day.png',
        bgNight:       'PixelArt/Paralax/Layer1/night.png',
      };
      let pending = Object.keys(toLoad).length;
      for (const [key, src] of Object.entries(toLoad)) {
        const img = new Image();
        img.onload = () => { if (--pending === 0) onComplete(); };
        img.onerror = () => console.error('Failed to load sprite:', src);
        img.src = src;
        sprites[key] = img;
      }
    }

    window.addEventListener('load', () => {
      loadSprites(() => {
        requestAnimationFrame(gameLoop);
      });
    });

Place loadSprites() in a new file `src/sprites.js` and add it to index.html before main.js.
